git常用命令

//克隆远程仓库
git clone [地址]

// 初始化 在工作路径上创建主分支
git init

// 克隆远程仓库
git clone 地址

// 克隆分支的代码到本地
git clone -b 分支名 地址

// 将某个文件存入暂存区
git add 文件名

//把b和c存入暂存区
git add b c

// 将所有文件提交到暂存区
git add .

// 一个文件分多次提交
git add -p 文件名

// 提交到仓库
git commit -m “提交的备注信息”

// 把指定标签推送到远程仓库
git push origin 标签名

// 新建分支
git branch 分支名

// 查看当前所有分支
git branch

// 检出分支
git checkout 分支名

// 创建并切换分支
git checkout -b 分支名

// 查看分支以及提交hash值和commit信息
git branch -v

// 把该分支的内容合并到现有分支上
git merge 分支名

// 删除分支
git branch -d 分支名

// 修改分支名
git branch -m 旧分支名 新分支名

// 列出远程分支(远程所有分支名)
git branch -r

// 查看远程分支(列出远程分支以及本地分支名)
git branch -a

// 更新remote索引
git fetch

// 将本地分支推送到origin主机
git push -u origin 分支名
